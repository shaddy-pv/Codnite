# Curl Format File for Performance Testing

# This file defines the format for curl output when measuring performance
# Usage: curl -w "@curl-format.txt" -o /dev/null -s http://localhost/api/health

# Connection time (time to establish TCP connection)
time_namelookup:  %{time_namelookup}\n

# Time to connect to the server
time_connect:     %{time_connect}\n

# Time to establish SSL connection (if HTTPS)
time_appconnect:  %{time_appconnect}\n

# Time to establish TCP connection to the server
time_pretransfer: %{time_pretransfer}\n

# Time to start the transfer of the response
time_redirect:    %{time_redirect}\n

# Time to start the transfer of the response
time_starttransfer: %{time_starttransfer}\n

# Total time for the request
time_total:       %{time_total}\n

# HTTP response code
http_code:        %{http_code}\n

# Size of the response
size_download:    %{size_download}\n

# Size of the request
size_upload:      %{size_upload}\n

# Size of the headers
size_header:      %{size_header}\n

# Speed of the download
speed_download:   %{speed_download}\n

# Speed of the upload
speed_upload:     %{speed_upload}\n

# Content type of the response
content_type:     %{content_type}\n

# Effective URL (after redirects)
url_effective:    %{url_effective}\n

# Remote IP address
remote_ip:        %{remote_ip}\n

# Remote port
remote_port:      %{remote_port}\n

# Local IP address
local_ip:         %{local_ip}\n

# Local port
local_port:       %{local_port}\n

# Number of redirects
num_redirects:    %{num_redirects}\n

# Number of connections
num_connects:     %{num_connects}\n

# SSL verify result
ssl_verify_result: %{ssl_verify_result}\n

# HTTP version
http_version:     %{http_version}\n

# Request method
http_method:      %{http_method}\n

# Request URL
http_url:         %{http_url}\n

# Response headers
response_headers: %{response_headers}\n

# Request headers
request_headers:  %{request_headers}\n

# Response body
response_body:     %{response_body}\n

# Request body
request_body:     %{request_body}\n

# Error message
error_message:    %{error_message}\n

# Error code
error_code:       %{error_code}\n

# Exit code
exit_code:        %{exit_code}\n

# New line for readability
\n
